    Parameters

    • options: {
          dirpath: string;
          format?: GameFormat | Iterable<GameFormat, any, any>;
          type?: CardDB.File.DBType | Iterable<CardDB.File.DBType, any, any>;
          walk?: boolean;
      }

      Options.

      • dirpath: string

        Directory path to load.

      • Optionalformat?: GameFormat | Iterable<GameFormat, any, any>

        Match exact game format of a sorted_format CardDB.

      • Optionaltype?: CardDB.File.DBType | Iterable<CardDB.File.DBType, any, any>

        Match type of CardDB.

      • Optionalwalk?: boolean

        Walk all subdirectories for CardDB files to load; default: false

    Returns Promise<CardDB.Stream.Reader[]>
