Interface CardSorted

Additional data added to cards when sorting.

interface CardSorted {
    border_color: string;
    card_faces?: CardFace[];
    cmc: number;
    collector_number: string;
    color_identity: Colors;
    colors: Colors;
    defense?: string;
    filename: string;
    foil: string;
    game_changer: boolean;
    keywords: string[];
    lang: string;
    lang_csv?: string;
    legalities: Record<string, string>;
    loyalty?: string;
    mana_cost: string;
    mark?: "error" | "ok" | "warning";
    name: string;
    object: "card";
    oracle_id: string;
    oracle_text: string;
    power?: string;
    price: string;
    printed_name?: string;
    produced_mana?: Colors;
    quantity: number;
    rarity: string;
    rarity_orig: string;
    rarity_recent: string;
    released_at: string;
    reserved: boolean;
    scryfall_id: string;
    scryfall_uri: string;
    set: string;
    set_name: string;
    set_type: string;
    toughness?: string;
    type: string;
    type_line: string;
}

mana_cost: string

The mana cost for this card. This value will be any empty string "" if the cost is absent. Remember that per the game rules, a missing mana cost and a mana cost of {0} are different values. Multi-faced cards will report this value in card faces.
