Interface CardFace

Defines the enriched, but reduced set of card face data transferred from the Scryfall DB.

See

https://scryfall.com/docs/api/cards

interface CardFace {
    cmc?: number;
    colors?: Colors;
    defense?: string;
    loyalty?: string;
    mana_cost: string;
    name: string;
    object: "card_face";
    oracle_text?: string;
    power?: string;
    toughness?: string;
    type?: string;
    type_line?: string;
}
